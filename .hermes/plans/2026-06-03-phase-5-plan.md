### Task 1: Add document detail endpoint to backend

**Objective:** Backend endpoint to fetch a single document with full details (OCR text, validation results, metadata).

**Files:**
- Create: `backend/controllers/adminController.js` (add `getDocument` function)
- Modify: `backend/services/adminService.js` (add `getDocument` query)
- Modify: `backend/routes/admin.js` (add `GET /documents/:id`)

**Implementation:**

In `adminService.js`, add:
```javascript
async function getDocument(id) {
  const result = await pool.query(
    `SELECT md.*, a.first_name, a.last_name, a.email,
            sp.business_name
     FROM mechanic_documents md
     INNER JOIN service_provider_profiles sp ON sp.id = md.provider_id
     INNER JOIN accounts a ON a.id = sp.account_id
     WHERE md.id = $1`,
    [id]
  );
  if (!result.rows[0]) throw createError(404, 'Document not found');
  return sanitize(result.rows[0]);
}
```

In `adminController.js`, add:
```javascript
async function getDocument(req, res) {
  const data = await adminService.getDocument(req.params.id);
  return res.status(200).json({ data });
}
```

In `routes/admin.js`, add:
```javascript

```

In `frontend/src/api/admin.js`, add:
```javascript

  return client.get(`/admin/documents/${id}`);

```

### Task 1: Add provider availability slots table + migration

**Objective:** Replace the simple `is_available` boolean with a proper availability schedule.

**Files:**
- Create: `backend/db/migrations/006_provider_availability.sql`

```sql
-- Provider availability windows: each row = one recurring availability window
CREATE TABLE IF NOT EXISTS provider_availability (
  id BIGSERIAL PRIMARY KEY,
  provider_id INT NOT NULL REFERENCES service_provider_profiles(id) ON DELETE CASCADE,
  day_of_week INT NOT NULL CHECK (day_of_week BETWEEN 0 AND 6), -- 0=Sun, 6=Sat
  start_time TIME NOT NULL,
  end_time TIME NOT NULL,
  is_active BOOLEAN NOT NULL DEFAULT TRUE,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Provider date exceptions: specific dates where provider is unavailable
CREATE TABLE IF NOT EXISTS provider_availability_exceptions (
  id BIGSERIAL PRIMARY KEY,
  provider_id INT NOT NULL REFERENCES service_provider_profiles(id) ON DELETE CASCADE,
  exception_date DATE NOT NULL,
  is_available BOOLEAN NOT NULL DEFAULT FALSE,
  reason VARCHAR(255),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE (provider_id, exception_date)
);
```

### Task 1: Create WhatsApp service

**Objective:** Service module that sends WhatsApp messages via WhatsApp Cloud API.

**Files:**
- Create: `backend/services/whatsappService.js`
- Modify: `backend/config/index.js` — add WhatsApp config

```javascript
// backend/services/whatsappService.js
const axios = require('axios');
const config = require('../config');
const logger = require('../utils/logger');

const WHATSAPP_API = 'https://graph.facebook.com/v18.0';
const PHONE_NUMBER_ID = config.whatsapp.phoneNumberId;
const TOKEN = config.whatsapp.token;

async function sendMessage(to, body) {
  const response = await axios.post(
    `${WHATSAPP_API}/${PHONE_NUMBER_ID}/messages`,
    {
      messaging_product: 'whatsapp',
      to,
      type: 'text',
      text: { body },
    },
    {
      headers: {
        Authorization: `Bearer ${TOKEN}`,
        'Content-Type': 'application/json',
      },
    }
  );
  return response.data;
}

async function sendBookingConfirmation(customerPhone, bookingData) {
  const message = `🔧 *Booking Confirmed!*\n\n`
    + `Booking #${bookingData.id}\n`
    + `Provider: ${bookingData.business_name}\n`
    + `Date: ${new Date(bookingData.preferred_schedule).toLocaleDateString()}\n`
    + `Description: ${bookingData.description}\n\n`
    + `Thank you for choosing OneStopShop!`;
  return sendMessage(customerPhone, message);
}

async function sendNewBookingNotification(providerPhone, bookingData) {
  const message = `🔔 *New Booking Request!*\n\n`
    + `Booking #${bookingData.id}\n`
    + `Customer: ${bookingData.customer_name}\n`
    + `Date: ${new Date(bookingData.preferred_schedule).toLocaleDateString()}\n`
    + `Description: ${bookingData.description}\n\n`
    + `Check your dashboard to accept or reschedule.`;
  return sendMessage(providerPhone, message);
}

module.exports = { sendMessage, sendBookingConfirmation, sendNewBookingNotification };
```
